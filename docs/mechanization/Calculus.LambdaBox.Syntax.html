{-# OPTIONS --safe #-}
module Calculus.LambdaBox.Syntax where

open import Data.Nat using (ℕ)
open import Data.List using (List)

data Type : Set where
  `⊤   : Type
  _`→_ : Type → Type → Type
  `□_  : Type → Type

data Term : Set where
  `unit         : Term

  `box          : Term → Term
  `let-box_`in_ : Term → Term → Term

  `λ⦂_∙_        : Type → Term → Term
  _`$_          : Term → Term → Term

  `#¹_          : ℕ → Term
  `#⁰_          : ℕ → Term

Context : Set
Context = List Type
